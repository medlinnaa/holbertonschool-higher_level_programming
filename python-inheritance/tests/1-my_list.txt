# check print_sorted() with sorted append
>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.print_sorted()
[1, 2, 3]
>>> my_list
[1, 2, 3]

# check print_sorted() with not sorted append
>>> my_list = MyList()
>>> my_list.append(3)
>>> my_list.append(1)
>>> my_list.append(2)
>>> my_list.print_sorted()
[1, 2, 3]
>>> my_list
[3, 1, 2]

# check print_sorted() with negative numbers
>>> my_list = MyList()
>>> my_list.append(-1)
>>> my_list.append(3)
>>> my_list.append(-5)
>>> my_list.print_sorted()
[-5, -1, 3]
>>> my_list
[-1, 3, -5]

# check print_sorted() with empty list
>>> my_list = MyList()
>>> my_list.print_sorted()
[]
>>> my_list
[]

# check print_sorted() returns a new list (does not modify original)
>>> my_list = MyList([4, 2, 1, 3])
>>> my_list.print_sorted()
[1, 2, 3, 4]
>>> my_list
[4, 2, 1, 3]
